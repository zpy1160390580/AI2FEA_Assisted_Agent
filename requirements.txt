# ============================================================================
# AI2FEA 辅助智能体 - Python 依赖配置
# ============================================================================
# 项目：基于 AI 的 Abaqus 有限元分析自动化系统
# 技术栈：LlamaIndex + 硅基流动平台 (DeepSeek-V3) + Streamlit
# Python 版本要求：3.8+（推荐 3.11）
# ============================================================================

# ----------------------------------------------------------------------------
# 核心框架：LlamaIndex 智能体框架
# ----------------------------------------------------------------------------
llama-index>=0.10.0,<0.12.0            # LlamaIndex 核心库
llama-index-core>=0.10.0,<0.12.0       # LlamaIndex 核心组件
llama-index-llms-openai>=0.1.0         # OpenAI 兼容的 LLM 接口

# ----------------------------------------------------------------------------
# Web 应用：Streamlit 交互界面
# ----------------------------------------------------------------------------
streamlit>=1.36.0,<2.0.0               # Web 应用框架

# ----------------------------------------------------------------------------
# 环境配置：环境变量管理
# ----------------------------------------------------------------------------
python-dotenv>=1.0.0,<2.0.0            # 从 .env 文件加载环境变量

# ----------------------------------------------------------------------------
# AI 模型接口：OpenAI 兼容 API
# ----------------------------------------------------------------------------
openai>=1.0.0,<2.0.0                   # OpenAI SDK（支持硅基流动平台）

# ----------------------------------------------------------------------------
# 数据处理：Pandas 数据分析
# ----------------------------------------------------------------------------
pandas>=2.0.0,<3.0.0                   # 数据处理和评估结果分析

# ----------------------------------------------------------------------------
# HTTP 客户端：网络请求
# ----------------------------------------------------------------------------
httpx>=0.25.0,<0.28                    # HTTP 客户端（版本限制以确保兼容性）

# ----------------------------------------------------------------------------
# 异步支持：事件循环管理
# ----------------------------------------------------------------------------
nest-asyncio>=1.5.0                    # 解决 Jupyter/Streamlit 中的异步事件循环冲突

# ----------------------------------------------------------------------------
# 数据序列化：pyarrow（必须先于 phoenix 指定，避免被拉取到无 Windows wheel 的旧版本 10.0.0）
# ----------------------------------------------------------------------------
pyarrow>=15.0.0,<19.0.0                # Apache Arrow（Phoenix/Pandas 依赖，新版本提供 Windows wheel）

# ----------------------------------------------------------------------------
# 可观测性和评估：Phoenix 追踪平台（简化版，避免依赖冲突）
# ----------------------------------------------------------------------------
arize-phoenix>=14.0.0                  # Phoenix 核心库
arize-phoenix-evals>=2.11.0,<3.0.0    # Phoenix 评估工具（TOOL_CALLING_PROMPT_RAILS_MAP, OpenAIModel, llm_classify）
openinference-instrumentation-llama-index>=2.0.0,<3.0.0  # LlamaIndex 追踪插桩

# ============================================================================
# 硅基流动平台兼容性说明
# ============================================================================
# llama-index-llms-openai 支持通过 api_base 参数连接兼容 OpenAI API 的服务
# 硅基流动平台提供完全兼容 OpenAI 的 API 接口，无需额外依赖
#
# 配置示例（在 config_files/config.py 中）：
#   SILICONFLOW_BASE_URL = "https://api.siliconflow.cn/v1"
#   MODEL_NAME = "deepseek-ai/DeepSeek-V3"
#
# 支持的模型：
#   - deepseek-ai/DeepSeek-V3 (通用任务，性能强，成本低)
#   - deepseek-ai/DeepSeek-R1 (复杂推理，数学计算)
#   - Qwen/Qwen2.5-72B-Instruct (阿里通义千问，中文友好)
# ============================================================================

# ============================================================================
# Abaqus 依赖说明
# ============================================================================
# Abaqus Python 脚本（FEA_scripts/*.py）使用 Abaqus 内置的 Python 环境
# 以下模块由 Abaqus 提供，无需在此安装：
#   - abaqus, abaqusConstants (Abaqus CAE API)
#   - odbAccess (ODB 文件读取)
#   - regionToolset (区域工具集)
#
# 注意：运行本项目需要本地安装 Abaqus 并配置环境变量
# ============================================================================

# ============================================================================
# 安装说明
# ============================================================================
# 方式 1：一键安装（推荐）
#    pip install -r requirements.txt
#
# 方式 2：分步安装（如遇到依赖冲突）
#    # 先安装核心依赖
#    pip install llama-index llama-index-core llama-index-llms-openai
#    pip install streamlit python-dotenv openai pandas httpx nest-asyncio
#
#    # 再安装 Phoenix（可选，用于可观测性）
#    pip install arize-phoenix openinference-instrumentation-llama-index
#
# 完整流程：
# 1. 创建 Conda 环境：
#    conda create -n AI2FEA python=3.11 -y
#    conda activate AI2FEA
#
# 2. 安装依赖：
#    pip install -r requirements.txt
#
# 3. 配置 API Key：
#    cd config_files
#    cp .env.example .env
#    # 编辑 .env 文件，填入硅基流动 API Key
#
# 4. 运行应用：
#    streamlit run main.py
# ============================================================================
